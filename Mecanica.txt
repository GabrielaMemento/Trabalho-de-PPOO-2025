/**
 * Define se este animal pode comer outro objeto.
 */
public boolean canEat(Object obj) {
    return false; // padrão: não come nada
}

//coelho
@Override
public boolean canEat(Object obj) {
    return obj instanceof Planta;
}

//RAPOSA
@Override
public boolean canEat(Object obj) {
    return obj instanceof Rabbit;
}

// lobo
@Override
public boolean canEat(Object obj) {
    return (obj instanceof Rabbit) || (obj instanceof Fox);
}

//aguia
@Override
public boolean canEat(Object obj) {
    return (obj instanceof Rabbit) || (obj instanceof Cobra);
}

//cobra
@Override
public boolean canEat(Object obj) {
    // Cobra pode comer qualquer Animal, exceto o Caçador
    return (obj instanceof Animal) && !(obj instanceof Cacador);
}

//caçafor
@Override
public boolean canEat(Object obj) {
    // Caçador pode atacar qualquer animal, exceto águias e cobras
    return (obj instanceof Animal) 
           && !(obj instanceof Aguia) 
           && !(obj instanceof Cobra);
}
